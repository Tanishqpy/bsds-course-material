r
# Load necessary libraries
library(tidyr)
library(dplyr)

# Read the data
measles <- read.csv("measles.csv", stringsAsFactors = FALSE)

# Reshape the data so that each row is a year and each column is a state.
# The "rate" values become the cell entries.
measles_wide <- measles %>%
  pivot_wider(names_from = state, values_from = rate)

# Log-transform the rates (excluding the 'year' column).
# (Note: Ensure that all rate values are positive. If there are zeros, you'll need to adjust before taking the log.)
log_rates <- log(measles_wide[,-1])

# Compute the correlation matrix using pairwise complete observations
cor_mat <- cor(log_rates, use = "pairwise.complete.obs")

# Remove self-correlations by setting the diagonal to NA
diag(cor_mat) <- NA

# Find the maximum correlation (most similar pair)
max_corr <- max(cor_mat, na.rm = TRUE)
max_pair <- which(cor_mat == max_corr, arr.ind = TRUE)[1, ]  # take the first pair if multiple

# Find the minimum correlation (least similar pair)
min_corr <- min(cor_mat, na.rm = TRUE)
min_pair <- which(cor_mat == min_corr, arr.ind = TRUE)[1, ]  # take the first pair if multiple

# Get the corresponding state names from the column names of log_rates
state_names <- colnames(log_rates)

cat("Most similar pair (highest correlation):\n")
cat(state_names[max_pair[1]], "and", state_names[max_pair[2]], "with correlation", max_corr, "\n\n")

cat("Least similar pair (lowest correlation):\n")
cat(state_names[min_pair[1]], "and", state_names[min_pair[2]], "with correlation", min_corr, "\n")

# Identify the pair with the highest correlation (most similar).
